{{- $classes := "" -}} {{- $has_top := or (ne nil .mt) (ne nil .Params.mt) -}} {{- $has_bottom := or (ne nil .mb) (ne nil .Params.mb) -}} {{- if $has_top -}} {{- $top := default .Params.mt .mt -}} {{- $classes = printf `%s mt-%s` $classes $top -}} {{- end -}} {{- if $has_bottom -}} {{- $bottom := default .Params.mb .mb -}} {{- $classes = printf `%s mb-%s` $classes $bottom -}} {{- end -}} {{ return $classes }}
